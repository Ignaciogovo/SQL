1 Consultar tablas que hay.
SELECT * FROM AdventureWorks2017.INFORMATION_SCHEMA.TABLES
2 Cuantas tablas hay:
SELECT COUNT(*) FROM AdventureWorks2017.INFORMATION_SCHEMA.TABLES
3 Cuantos campos tiene la tabla Person.Person
SELECT COUNT(*) FROM  INFORMATION_SCHEMA.COLUMNS 
WHERE TABLE_NAME= 'Person'
AND TABLE_SCHEMA = 'Person'

4 Numero de campos ordenados de mayor a menor de todas las tablas
SELECT TABLE_NAME, COUNT(*) AS "numero" FROM  INFORMATION_SCHEMA.COLUMNS 
GROUP BY(TABLE_NAME)
ORDER BY "numero" DESC
5 La tabla con mas campos 
SELECT TOP(1)TABLE_NAME, COUNT(*) AS numero
FROM  INFORMATION_SCHEMA.COLUMNS 
GROUP BY(TABLE_NAME)
ORDER BY numero DESC
