CREATE PROCEDURE vermenu @tipoalimento INT, @categoria INT
AS
	IF @categoria is null
		select * FROM menu where tipo = @tipoalimento
		else
		select * FROM menu WHERE tipo = @tipoalimento AND categoria = @categoria;
GO
