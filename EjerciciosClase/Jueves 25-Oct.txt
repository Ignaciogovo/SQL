1 Crear un procedimiento almacenado que nos devuelva el factorial de un número​
	USE arepazo
	DROP PROC factorial;
	go
	CREATE PROCEDURE factorial @numero INT
	as
	declare @resultado int =1
	while @numero > 0
	begin 
		set @resultado=@resultado*@numero;
		set @numero=@numero-1;
	end
	print @resultado
	go
	
	exec factorial 4


2 Crear un procedimiento almacenado que nos indique si un numero es primo o no.​
USE arepazo
go
CREATE or alter PROCEDURE primo @i INT
as
declare @numero as int = @i-1;
declare @resto as int;
while @numero >= 2
BEGIN
	set @resto=@i%@numero;
	if @resto = 0
	BEGIN
		print 'No es primo';
		print 'Es divisible por';
		print @numero;
	END
		set @numero=@numero-1;
END
if @resto <> 0
	BEGIN
		print 'Es primo';
	END
go
	
	exec primo 28


3 Crear un procedimiento almacenado que nos devuelva los n primeros numeros de la sucesión de fibonacci.

USE arepazo
go
CREATE or alter PROCEDURE fibonacci @i INT
as
	declare @numero1 int = 1;
	declare @numero2 int = 1;
	declare @numero3 int;
	while @numero1<=@i
	begin
	print @numero1
	set @numero3=@numero2
	set @numero2=@numero2+@numero1;
	set @numero1=@numero3;
	end
go
exec fibonacci 20
